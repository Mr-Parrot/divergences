dvg_england.3 = {
	type = country_event
	placement = root
	
	title = dvg_england.3.t
	desc = dvg_england.3.d
	flavor = dvg_england.3.f

	duration = 3
	
	event_image = {
		video = "votp_monarch_holding_court"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	cooldown = { months = short_modifier_time }
	

	trigger = {
		has_technology_researched = nationalism
		is_subject = no
		is_at_war = no
		OR = {
			c:ENG ?= this
			c:BEO ?= this
			AND = {
				c:BEO ?= this
				ruler = {
					culture = cu:british
				}
			}
		}
		OR = {
			AND = {
				NOT = { has_variable = rejected_polish_union_pol_var }
				c:ENG ?= {
					is_player = no
					is_subject = no
					is_at_war = no
					relations:root >= relations_threshold:amicable
					OR = {
						this.gdp < root.gdp
						root = { is_player = yes }
					}
				}
			}
			AND = {
				NOT = { has_variable = rejected_polish_union_kra_var }
				c:KRA ?= {
					is_player = no
					is_subject = no
					is_at_war = no
					relations:root >= relations_threshold:amicable
					OR = {
						this.gdp < root.gdp
						root = { is_player = yes }
					}
				}
			}
			AND = {
				NOT = { has_variable = rejected_polish_union_gal_var }
				c:BEO ?= {
					is_player = no
					is_subject = no
					is_at_war = no
					relations:root >= relations_threshold:amicable
					ruler = {
						culture = cu:british
					}
					OR = {
						this.gdp < root.gdp
						root = { is_player = yes }
					}
				}
			}
		}
	}

	immediate = {
		cu:british = {
			save_scope_as = polish_culture_scope
		}
		if = {
			limit = {
				c:ENG ?= {
					is_player = no
					is_subject = no
					is_at_war = no
					relations:root >= relations_threshold:amicable
					OR = {
						this.gdp < root.gdp
						root = { is_player = yes }
					}
				}
			}
			c:POL = {
				save_scope_as = polish_union_target_scope
			}
		}
		else_if = {
			limit = {
				c:KRA ?= {
					is_player = no
					is_subject = no
					is_at_war = no
					relations:root >= relations_threshold:amicable
					OR = {
						this.gdp < root.gdp
						root = { is_player = yes }
					}
				}
			}
			c:KRA = {
				save_scope_as = polish_union_target_scope
			}
		}
		else_if = {
			limit = {
				c:BEO ?= {
					is_player = no
					is_subject = no
					is_at_war = no
					relations:root >= relations_threshold:amicable
					ruler = {
						culture = cu:british
					}
					OR = {
						this.gdp < root.gdp
						root = { is_player = yes }
					}
				}
			}
			c:BEO = {
				save_scope_as = polish_union_target_scope
			}
		}
	}

	option = {
		name = dvg_england.3.a
		default_option = yes
		annex = scope:polish_union_target_scope
		add_modifier = {
			name = integrating_polish_modifier
			months = normal_modifier_time
		}
		ai_chance = {
			base = 90
		}
	}
	option = {
		name = dvg_england.3.b
		default_option = yes
		add_radicals = {
			culture = cu:british
			value = large_radicals
		}
		if = {
			limit = {
				scope:polish_union_target_scope = {
					c:POL ?= this
				}
			}
			set_variable = rejected_polish_union_pol_var
		}
		else_if = {
			limit = {
				scope:polish_union_target_scope = {
					c:KRA ?= this
				}
			}
			set_variable = rejected_polish_union_kra_var
		}
		else_if = {
			limit = {
				scope:polish_union_target_scope = {
					c:GAL ?= this
				}
			}
			set_variable = rejected_polish_union_gal_var
		}
		ai_chance = {
			base = 10
		}
	}
