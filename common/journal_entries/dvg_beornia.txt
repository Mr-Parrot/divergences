je_new_mercia_independent = {
	icon = "gfx/interface/icons/invention_icons/mechanized_farming.dds"

	group = je_group_foreign_affairs

	is_shown_when_inactive = {	
	exists = c:BEO
	this = c:BEO
	}

	possible = {
		has_global_variable = new_mercia
		owns_entire_state_region = STATE_NEAR_COLLASUYU
		NOT = { exists = c:NME  }
	}

	immediate = {
		set_variable = {
			name = new_mercia_var
			value = 0
		}
	}

	

	complete = {
		# If-else clause to avoid errors because of missing variable before JE activates
		ig:ig_rural_folk = {
				ig_approval < 0
				is_in_government = no
			}
		trigger_if = {
			limit = { exists = var:new_mercia_var }
			
			
		  OR = {
			scope:journal_entry = { is_goal_complete = yes }
			
			}
		}
		
		
		
	}
	
	on_complete = {
		
		set_variable = new_mercia_complete
		trigger_event = {
			id = dvg_beornia.22
		}
	}
	
	
	invalid = {
		exists = c:NME
	}
	on_invalid = {
		
	}
	

	

	current_value = {
		value = root.var:new_mercia_var
	}

	goal_add_value = {
		value = 50
	}

	progressbar = yes
	
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					ig:ig_rural_folk = {
				ig_approval < 0
			}
				}
				
				change_variable = {
					name = new_mercia_var
					add = 1
				}
				set_variable = new_mercia_subject
			}
			
			
			
		}
	}
	
	
	
	

	weight = 10000
	should_be_pinned_by_default = yes
}
